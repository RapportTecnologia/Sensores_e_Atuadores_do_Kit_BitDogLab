47// Patterned after CMSIS NVIC_SystemReset
48__attribute__((__noreturn__)) static inline void system_reset() {
49 __DSB(); /* Ensure all outstanding memory accesses included
50 buffered write are completed before reset */
51 scb_hw->aircr = ((0x5FAUL << 16U) | (1UL << 2U));
52 __DSB(); /* Ensure completion of memory access */
53 for (;;) {
54 __asm volatile("nop");
55 }
56}
